#version 400 core

in vec3 position;
in vec2 textureCoord;

uniform mat4 viewProjectionMatrix;
uniform mat4 worldTransformMatrix;

out vec2 pass_textureCoords;

void main()
{
    gl_Position = viewProjectionMatrix * worldTransformMatrix * vec4(position, 1.0);
    pass_textureCoords = textureCoord;
}
