#version 400 core

out vec4 out_Color;

in vec3 pass_textureCoords;

in float lightValue;

uniform sampler2DArray textureSampler;

void main()
{ 
    out_Color = texture(textureSampler, pass_textureCoords);
    if(out_Color.a < 0.5)
    {
        discard;
    }

    out_Color = lightValue * out_Color;
}
